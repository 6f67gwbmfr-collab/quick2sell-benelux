export default async function handler(req, res) { if (req.method !== "POST") { return res.status(405).json({ error: "Method not allowed" }); } const { info } = req.body; const response = await fetch("https://api.openai.com/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${process.env.OPENAI_API_KEY}`, "Content-Type": "application/json" }, body: JSON.stringify({ model: "gpt-4.1-mini", messages: [{ role: "user", content: `Analyseer dit tweedehands product en geef: 1. Productnaam 2. Geschatte waarde 3. Snelle verkoopprijs 4. Maximale vraagprijs 5. Beste verkoopplatforms 6. Verkooptips Product: ${info}` }] }) }); const data = await response.json(); res.status(200).json({ result: data.choices[0].message.content }); }
